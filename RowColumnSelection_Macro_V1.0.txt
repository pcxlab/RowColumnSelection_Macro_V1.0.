Sub FormatDateAndEditCells()
    ' This macro formats the cells in the selected column starting from the active cell,
    ' changing their date format to "dd-mm-yyyy" and entering Edit Mode (F2) before pressing Enter.
    ' The loop only runs from the active cell row to the last row with data in the selected column.
    
    ' Spedup process by disabling screen update
    
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    
    Dim lastRow As Long          ' Last row with data in the selected column
    Dim selectedColumn As Long   ' Column of the active cell
    Dim ws As Worksheet          ' Active worksheet
    Dim cell As Range            ' Variable for each cell in the range
    Dim activeRow As Long        ' Row of the active cell
    
    ' Set the active worksheet
    Set ws = ActiveSheet
    
    ' Get the column number of the selected cell
    selectedColumn = Selection.Column
    
    ' Find the last row with data in the selected column
    lastRow = ws.Cells(ws.Rows.Count, selectedColumn).End(xlUp).Row
    
    ' Get the row number of the active cell
    activeRow = ActiveCell.Row
    
    ' Ensure the loop starts from the active row and runs to the last row with data
    If activeRow <= lastRow Then
        ' Loop through each cell from the active row to the last row with data
        For Each cell In ws.Range(ws.Cells(activeRow, selectedColumn), ws.Cells(lastRow, selectedColumn))
            ' Format the cell as a date in "dd-mm-yyyy" format
            'cell.NumberFormat = "dd-mm-yyyy"
            
            ' Simulate pressing F2 to enter Edit Mode
            Application.SendKeys "{F2}"
            
            ' Simulate pressing Enter to confirm the edit
            Application.SendKeys "{ENTER}"
        Next cell
    End If
    
    ' Reseting screenupdating options
    Application.ScreenUpdating = True
    Application.Calculation = xlCalculationAutomatic
    
    MsgBox "Completed"
        
End Sub
